    +THRUGHOST
	+MISSILEMORE
	+LOOKALLAROUND

	+BOSS
	Monster
	Radius 16
	Height 48
	Species "Totenkopf"
	Scale 1.05
	+NOINFIGHTING
	var int user_timer;
	States
	{
		Spawn:
			TK44 M 0

			TK44 M 1 A_CustomMeleeAttack(3,0,0,0,0)

		Revive:

			TK44 Q 0 A_RaiseMaster
			Stop
		Fade:
		Death:
		XDeath:
			TK44 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)
			TK44 N 0 A_NoBlocking
			TK44 N 6
